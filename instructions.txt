follow below steps to activate virtual env for this project
    cd ..
    .\venv\Scripts\activate
to start project
    cd ecomweb
    python ./manage.py runserver
to migrate db
    python ./manage.py migrate
    python ./manage.py makemigrations
to install a package
    pip install PACKAGE_NAME
static files
    python manage.py collectstatic 
create new app
    python manage.py startapp APP_NAME

git commands

git init
git add . -- add all new and modified files to staging area to be commited to repository
git commit -m "Initial commit" -- create snapshot of changes made
git pull
git remote add origin <project url>
git push -f origin <branch name>
git rev-parse --abbrev-ref HEAD -- check current active branch
